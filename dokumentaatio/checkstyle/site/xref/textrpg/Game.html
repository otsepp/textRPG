2   package textrpg;
3   
4   import java.util.ArrayList;
5   import java.util.List;
6   import java.util.Scanner;
7   import textrpg.characters.Enemy;
8   import textrpg.characters.Player;
9   import textrpg.command.Command;
10  import textrpg.events.battle.BattleEvent;
11  import textrpg.events.ending.EndingEvent;
12  import textrpg.event.GameEvent;
13  import textrpg.event.GameEventReturnValues;
14  import textrpg.events.straightpath.StraightPathEvent;
15  
16  public class Game {
17      
18      private Player player;
19      private List<GameEvent> events;
20      private Scanner s;
21      
22      public Game() {
23          player = new Player("Player");
24          events = new ArrayList();
25          s = new Scanner(System.in);
26      }
27      
28      public void start() {
29          System.out.println("Welcome...\n");
30          createEvents();
31          
32          for (GameEvent e : events) {
33              boolean playerSurvives = handleEvent(e);
34              System.out.println("");
35              
36              if (!playerSurvives) {
37                  System.out.println("Game over!");
38                  break;
39              }
40          }
41      }
42      
43      public boolean handleEvent(GameEvent e) {
44          System.out.println("****" + e.getStartMessage() + "****");
45          
46          boolean continueEvent = true;
47          while (continueEvent) {
48              
49              List<Command> commands = e.getCommands();
50              for (int i = 0; i < commands.size(); i++) {
51                  System.out.println((i + 1) + ". " + commands.get(i).getDescription());
52              }
53              System.out.println("");
54              continueEvent = chooseAndExecuteCommand(e, commands);
55              System.out.println("");
56          }
57          
58          if (player.getHealth() > 0) {
59              return true;
60          } else {
61              return false;
62          }
63      }
64      
65      public boolean chooseAndExecuteCommand(GameEvent e, List<Command> commands) {
66          while (true) {
67              System.out.print("Choose command: ");
68              int commandId;
69              
70              if (s.hasNextInt()) {
71                  commandId = s.nextInt();
72              } else {
73                  System.out.println("Please enter a number.");
74                  s.next();
75                  continue;
76              }
77              GameEventReturnValues ret = e.initiateEvent(commandId - 1);
78              
79              if (ret != null) {
80                  System.out.println("");
81                  for (String msg : ret.getMessages()) {
82                      System.out.println(msg);
83                  }
84                  return ret.getEventContinues();
85              
86              } else {
87                  System.out.println("Invalid command id.");
88              }
89          }
90      }
91      
92      public void createEvents() {
93          this.events.add(new StraightPathEvent());
94          this.events.add(new BattleEvent(this.player, new Enemy("Bandit")));
95          this.events.add(new EndingEvent());
96      }
97  
98  }
