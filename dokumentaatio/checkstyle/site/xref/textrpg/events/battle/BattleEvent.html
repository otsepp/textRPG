2   package textrpg.events.battle;
3   
4   import java.util.ArrayList;
5   import java.util.List;
6   import textrpg.characters.Enemy;
7   import textrpg.characters.Player;
8   import textrpg.command.Command;
9   import textrpg.event.GameEvent;
10  import textrpg.event.GameEventReturnValues;
11  
12  public class BattleEvent extends GameEvent {
13  
14      private Player player;
15      private Enemy enemy;
16      private boolean playerTurn;
17      
18      private List<Command> defaultCommands;
19      
20      public BattleEvent(Player player, Enemy enemy) {
21          super("", new ArrayList());
22          this.player = player;
23          this.enemy = enemy;
24          super.startMessage = "You find yourself facing a " + enemy.getName();
25          this.playerTurn = true;
26          setDefaultCommands();
27      }
28  
29      
30      public boolean isPlayerTurn() {
31          return this.playerTurn;
32      }
33  
34      public void setPlayerTurn(boolean newValue) {
35          this.playerTurn = newValue;
36      }
37      
38      @Override
39      public final void setDefaultCommands() {
40          if (super.commands.size() > 0) {
41              super.commands.clear();
42          }
43          super.commands.add(new Attack(this));
44          super.commands.add(new SelectItem(this));
45      }
46      
47      @Override
48      public GameEventReturnValues initiateEvent(int commandId) {
49          GameEventReturnValues returnValues = super.initiateEvent(commandId);
50          
51          if (returnValues == null) {
52              return null;
53          }
54          
55          List<String> messages = returnValues.getMessages();
56          
57          if (super.commands != null && !this.playerTurn) {
58              attackPlayer(returnValues);
59          } else if (super.commands == null) {
60              messages.add(this.enemy.getName() + " is dead. You continue your way.");
61          }
62          return returnValues;
63      }
64      
65      private void attackPlayer(GameEventReturnValues returnValues) {
66          List<String> messages = returnValues.getMessages();
67  
68          int damage = this.enemy.getBaseDamage();
69          this.player.takeDamage(damage);
70          messages.add(this.enemy.getName() + " attacks " + this.player.getName() + " for " + damage + " damage.");
71          
72          if (this.player.getHealth() > 0) {
73              messages.add(this.player.getName() + " has " + this.player.getHealth() + " health remaining");
74          } else {
75              messages.add(this.player.getName() + " has died.");
76              super.commands = null;
77              returnValues.setEventContinues(false);
78          }
79      }
80     
81      public Player getPlayer() {
82          return this.player;
83      }
84      
85      public Enemy getEnemy() {
86          return this.enemy;
87      }
88      
89  }
