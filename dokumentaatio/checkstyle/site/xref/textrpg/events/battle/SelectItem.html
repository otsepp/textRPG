2   package textrpg.events.battle;
3   
4   import java.util.ArrayList;
5   import java.util.Map;
6   import textrpg.command.Command;
7   import textrpg.command.CommandReturnValues;
8   import textrpg.items.Item;
9   import textrpg.items.Usable;
10  
11  public class SelectItem extends Command {
12      private BattleEvent battle;
13      
14      public SelectItem(BattleEvent battle) {
15          super("Select an item", //description  
16                  new ArrayList(),    //messages
17                  battle.getCommands());   //new commands
18          this.battle = battle;
19      }
20  
21      
22      @Override
23      public CommandReturnValues executeCommand() {
24          battle.setPlayerTurn(true);
25          super.messages.clear();
26          
27          Map<Usable, Integer> usables = this.battle.getPlayer().getInventory().getUsableItems();
28          if (!usables.isEmpty()) {
29              setCommands(usables);
30          } else {
31              super.messages.add("No items to use!");
32          }
33          return new CommandReturnValues(super.messages, super.newCommands);
34      }
35      
36      private void setCommands(Map<Usable, Integer> usables) {
37          super.newCommands.clear();
38          for (Usable u : usables.keySet()) {
39              Item uAsItem = (Item) u;
40              super.newCommands.add(new UseItem(this.battle, uAsItem, usables.get(u)));
41          }
42      }
43      
44      public BattleEvent getBattleEvent() {
45          return this.battle;
46      }
47  }
